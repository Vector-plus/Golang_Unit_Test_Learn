思想，使用前缀和存储每一个元素的值。在进行区间操作时只需要操作区间的两个端点即可完成。

**一维差分**
```go
package main

import (
	"bufio"
	"fmt"
	"os"
)

func main() {
	in := bufio.NewReader(os.Stdin)

	var n, m, l, r, c int

	fmt.Fscanf(in, "%d %d\n", &n, &m)

	nums := make([]int, n)
	b := make([]int, n)

	n--
	for i := 0; i < n; i++ {
		fmt.Fscanf(in, "%d", &nums[i])
	}
	fmt.Fscanf(in, "%d\n", &nums[n])
	n++

	getB(nums, b)

	for i := 0; i < m; i++ {
		fmt.Fscanf(in, "%d %d %d\n", &l, &r, &c)
		insert(b, l-1, r-1, c)
	}

	prAns(b)

}

// 差分维护区间插入之后的数组值
func getB(nums, b []int) {
	b[0] = nums[0]
	for i := 1; i < len(nums); i++ {
		b[i] = nums[i] - nums[i-1]
	}
}

func insert(b []int, l, r, c int) {
	b[l] += c
	if r+1 < len(b) {
		b[r+1] -= c
	}
	// fmt.Println(b)
}

func prAns(b []int) {
	fmt.Printf("%d", b[0])
	for i := 1; i < len(b); i++ {
		b[i] += b[i-1]
		fmt.Printf(" %d", b[i])
	}
	fmt.Println()
}
```

**二维差分**
```go

package main

import (
	"bufio"
	"fmt"
	"os"
)

func main() {
	in := bufio.NewReader(os.Stdin)

	var n, m, t int

	fmt.Fscanf(in, "%d %d %d\n", &n, &m, &t)

	mp := make([][]int, n+10)
	bp := make([][]int, n+10)

	for i := 0; i < n; i++ {
		mp[i] = make([]int, m+10)
		bp[i] = make([]int, m+10)

		up := m - 1
		for j := 0; j < up; j++ {
			fmt.Fscanf(in, "%d", &mp[i][j])
		}
		fmt.Fscanf(in, "%d\n", &mp[i][up])
	}

	bp[n] = make([]int, m+10)

	getBP(mp, bp, n, m)

	var x1, y1, x2, y2, c int
	for i := 0; i < t; i++ {
		fmt.Fscanf(in, "%d %d %d %d %d\n", &x1, &y1, &x2, &y2, &c)
		insert(bp, x1-1, y1-1, x2-1, y2-1, c)
	}

	printMP(bp, n, m)
}

func getBP(mp, bp [][]int, n, m int) {
	for i := 0; i < n; i++ {
		if i == 0 {
			bp[0][0] = mp[0][0]
			for j := 1; j < m; j++ {
				bp[0][j] = mp[0][j] - mp[0][j-1]
			}
		} else {
			bp[i][0] = mp[i][0] - mp[i-1][0]
			for j := 1; j < m; j++ {
				bp[i][j] = mp[i][j] - mp[i-1][j] - mp[i][j-1] + mp[i-1][j-1]
			}
		}
	}
}

func insert(bp [][]int, x1, y1, x2, y2, c int) {
	bp[x1][y1] += c
	bp[x1][y2+1] -= c
	bp[x2+1][y1] -= c
	bp[x2+1][y2+1] += c
}

func printMP(bp [][]int, n, m int) {

	for i := 0; i < n; i++ {
		if i == 0 {
			fmt.Printf("%d", bp[0][0])
			for j := 1; j < m; j++ {
				bp[0][j] += bp[0][j-1]
				fmt.Printf(" %d", bp[0][j])
			}
			fmt.Println()
		} else {
			bp[i][0] += bp[i-1][0]
			fmt.Printf("%d", bp[i][0])
			for j := 1; j < m; j++ {
				bp[i][j] += bp[i-1][j] + bp[i][j-1] - bp[i-1][j-1]
				fmt.Printf(" %d", bp[i][j])
			}
			fmt.Println()
		}
	}
}
```
